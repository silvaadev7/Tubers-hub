        protectionEnabled = value
        
        if value then

            local Players = game:GetService("Players")
            local LocalPlayer = Players.LocalPlayer

            local blacklist = {
                {Name = "water", Class = "Part"},
            }

            local function neutralize(part)
                if part and part:IsA("BasePart") then
                    pcall(function()
                        part.Anchored = true
                        part.CanCollide = false
                        part.Massless = true
                        part.Transparency = 1
                        part:ClearAllChildren()
                    end)
                    pcall(function()
                        part:Destroy()
                    end)
                end
            end

            -- Conex√£o 1: DescendantAdded
            local conn1 = workspace.DescendantAdded:Connect(function(obj)
                for _, rule in ipairs(blacklist) do
                    if obj.Name == rule.Name and obj.ClassName == rule.Class then
                        neutralize(obj)
                    end
                end
            end)

            -- Limpeza inicial
            for _, obj in ipairs(workspace:GetDescendants()) do
                for _, rule in ipairs(blacklist) do
                    if obj.Name == rule.Name and obj.ClassName == rule.Class then
                        neutralize(obj)
                    end
                end
            end

            -- Conex√£o 2: Loop cont√≠nuo
            protectionLoop = task.spawn(function()
                while protectionEnabled and task.wait(0.25) do
                    for _, rule in ipairs(blacklist) do
                        for _, v in next, getnilinstances() do
                            if v.Name == rule.Name and v.ClassName == rule.Class then
                                neutralize(v)
                            end
                        end
                    end
                end
            end)

            -- Conex√£o 3: CharacterAdded
            local conn3 = LocalPlayer.CharacterAdded:Connect(function(char)
                local hum = char:WaitForChild("Humanoid")
                hum.Touched:Connect(function(hit)
                    for _, rule in ipairs(blacklist) do
                        if hit.Name == rule.Name and hit.ClassName == rule.Class then
                            neutralize(hit)
                        end
                    end
                end)
            end)

            -- Armazenar conex√µes
            table.insert(protectionConnections, conn1)
            table.insert(protectionConnections, conn3)
            
        else
            -- Desativar prote√ß√£o
            if protectionLoop then
                task.cancel(protectionLoop)
                protectionLoop = nil
            end
            
            for _, connection in ipairs(protectionConnections) do
                if connection then
                    connection:Disconnect()
                end
            end
            protectionConnections = {}
        end
    end
})

local function voidProtection(rootPart)
    if rootPart.Position.Y < -500 then
        local safeCFrame = CFrame.new(0, 100, 0)
        local rayParams = RaycastParams.new()
        rayParams.FilterDescendantsInstances = {LocalPlayer.Character}
        
        local result = Workspace:Raycast(rootPart.Position, Vector3.new(0, 500, 0), rayParams)
        rootPart.CFrame = result and CFrame.new(result.Position + Vector3.new(0, 5, 0)) or safeCFrame
    end
end

-- Controle das bolas de futebol
local function manageSoccerBalls()
    local soccerFolder = Workspace:FindFirstChild("Com", true)
                      and Workspace.Com:FindFirstChild("001_SoccerBalls")
    
    if soccerFolder then
        -- Atualiza bolas existentes
        for _, ball in ipairs(soccerFolder:GetChildren()) do
            if ball.Name:match("^Soccer") then
                pcall(function()
                    ball.CanCollide = not UltimateNoclip.Enabled
                    ball.Anchored = UltimateNoclip.Enabled
                end)
                UltimateNoclip.SoccerBalls[ball] = true
            end
        end
        
        -- Monitora novas bolas
        if not UltimateNoclip.Connections.BallAdded then
            UltimateNoclip.Connections.BallAdded = soccerFolder.ChildAdded:Connect(function(ball)
                if ball.Name:match("^Soccer") then
                    task.wait(0.3)
                    pcall(function()
                        ball.CanCollide = not UltimateNoclip.Enabled
                        ball.Anchored = UltimateNoclip.Enabled
                    end)
                end
            end)
        end
    end
end

-- Loop principal do sistema
local function mainLoop()
    UltimateNoclip.Connections.Heartbeat = RunService.Heartbeat:Connect(function()
        local character = LocalPlayer.Character
        
        -- Controle do jogador
        if character then
            managePlayerCollisions(character)
            
            local rootPart = character:FindFirstChild("HumanoidRootPart")
            if rootPart then
                voidProtection(rootPart)
            end
        end
        
        -- Atualiza bolas a cada 2 segundos
        if tick() % 2 < 0.1 then
            manageSoccerBalls()
        end
    end)
end

        telaGui.Parent = playerGui

        local function criarBotaoImagem(nome, posicao, idImagem, rotacao, acao)
            local botao = Instance.new("ImageButton")
            botao.Name = nome
            botao.Size = UDim2.new(0, 60, 0, 60)
            botao.Position = posicao
            botao.BackgroundTransparency = 1
            botao.Image = "rbxassetid://" .. idImagem
            botao.Rotation = rotacao
            botao.MouseButton1Down:Connect(acao)
            botao.Parent = telaGui
            return botao
        end

        -- Criar bot√µes de movimento
        local botaoFrente = criarBotaoImagem("BotaoFrente", UDim2.new(0, 10, 0.35, 0), "18478249606", 0, function()
            voando = true
            while voando and humanoidRootPart do
                humanoidRootPart.Velocity = humanoidRootPart.CFrame.LookVector * velocidade
                task.wait()
            end
        end)

        local botaoTras = criarBotaoImagem("BotaoTras", UDim2.new(0, 10, 0.5, 0), "18478249606", 180, function()
            voando = true
            while voando and humanoidRootPart do
                humanoidRootPart.Velocity = -humanoidRootPart.CFrame.LookVector * velocidade
                task.wait()
            end
        end)

        local botaoEsquerda = criarBotaoImagem("BotaoEsquerda", UDim2.new(1, -210, 0.3, 0), "18478249606", -90, function()
            voando = true
            while voando and humanoidRootPart do
                humanoidRootPart.Velocity = -humanoidRootPart.CFrame.RightVector * velocidade
                task.wait()
            end
        end)

        local botaoDireita = criarBotaoImagem("BotaoDireita", UDim2.new(1, -140, 0.3, 0), "18478249606", 90, function()
            voando = true
            while voando and humanoidRootPart do
                humanoidRootPart.Velocity = humanoidRootPart.CFrame.RightVector * velocidade
                task.wait()
            end
        end)

        -- Fun√ß√£o para parar o movimento
        local function pararVoo()
            voando = false
            if humanoidRootPart then
                humanoidRootPart.Velocity = Vector3.new(0, 0, 0)
            end
        end

        -- Conectar eventos de soltar bot√£o
        botaoFrente.MouseButton1Up:Connect(pararVoo)
        botaoTras.MouseButton1Up:Connect(pararVoo)
        botaoEsquerda.MouseButton1Up:Connect(pararVoo)
        botaoDireita.MouseButton1Up:Connect(pararVoo)

        -- Bot√£o Turbo (aumenta velocidade)
        local botaoTurbo = Instance.new("ImageButton")
        botaoTurbo.Name = "BotaoTurbo"
        botaoTurbo.Size = UDim2.new(0, 60, 0, 60)
        botaoTurbo.Position = UDim2.new(1, -130, 0, 10)
        botaoTurbo.BackgroundTransparency = 1
        botaoTurbo.Image = "rbxassetid://136213845438136"
        botaoTurbo.Parent = telaGui
        
        local turboAtivo = false
        botaoTurbo.MouseButton1Down:Connect(function()
            turboAtivo = not turboAtivo
            if turboAtivo then
                velocidade = 300
                botaoTurbo.ImageColor3 = Color3.fromRGB(255, 0, 0) -- Vermelho quando ativo
            else
                velocidade = 30
                botaoTurbo.ImageColor3 = Color3.fromRGB(255, 255, 255) -- Branco quando inativo
            end
        end)

        -- Bot√£o de minimizar
        local botaoMinimizar = Instance.new("TextButton")
        botaoMinimizar.Name = "BotaoMinimizar"
        botaoMinimizar.Size = UDim2.new(0, 60, 0, 60)
        botaoMinimizar.Position = UDim2.new(0, 10, 0, 10)
        botaoMinimizar.BackgroundTransparency = 1
        botaoMinimizar.Text = "-"
        botaoMinimizar.TextSize = 40
        botaoMinimizar.TextColor3 = Color3.fromRGB(255, 255, 255)
        botaoMinimizar.Parent = telaGui

        -- Lista de bot√µes para controlar visibilidade
        local botoes = {botaoFrente, botaoTras, botaoEsquerda, botaoDireita, botaoTurbo}
        local minimizado = false

        -- Fun√ß√£o para alternar visibilidade dos bot√µes
        local function alternarBotoes()
            minimizado = not minimizado
            for _, botao in ipairs(botoes) do
                botao.Visible = not minimizado
            end
            -- Alterar o texto do bot√£o de minimizar
            if minimizado then
                botaoMinimizar.Text = "+"
            else
                botaoMinimizar.Text = "-"
            end
        end

        botaoMinimizar.MouseButton1Down:Connect(alternarBotoes)

        -- Bot√£o de fechar (adicional)
        local botaoFechar = Instance.new("TextButton")
        botaoFechar.Name = "BotaoFechar"
        botaoFechar.Size = UDim2.new(0, 60, 0, 60)
        botaoFechar.Position = UDim2.new(0, 80, 0, 10)
        botaoFechar.BackgroundTransparency = 1
        botaoFechar.Text = "X"
        botaoFechar.TextSize = 30
        botaoFechar.TextColor3 = Color3.fromRGB(255, 0, 0)
        botaoFechar.Parent = telaGui
        
        botaoFechar.MouseButton1Down:Connect(function()
            telaGui:Destroy()
        end)

        print("Turbo V3 ativado! Use os bot√µes na tela para controlar o movimento.")
    end
})
